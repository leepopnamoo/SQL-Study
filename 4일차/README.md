``` 
create table "테이블명" ( 
칼럼명1  칼럼타입1 [Null or Not Null], 
칼럼명2  칼럼타입2 [Null or Not Null], 
칼럼명3  칼럼타입3 [Null or Not Null], 
칼럼명4  칼럼타입4 [Null or Not Null], 
...
칼럼명n  칼럼타입n [Null or Not Null], 
constraint "키이름" primary key ("칼럼명x", "칼럼명x" ...)
) ; 
```
